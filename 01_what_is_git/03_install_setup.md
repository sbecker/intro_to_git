# Installing Git #

    @@@ plain
    # Redhat / Fedora
    $ yum install git-core
    
    # Debian / Ubuntu
    $ apt-get install git-core
    
    # Mac OS X
    http://code.google.com/p/git-osx-installer
    
    # Windows
    http://code.google.com/p/msysgit

!SLIDE smaller

# First Time Setup #

    @@@ plain
    $ git config --global user.name "John Doe"
    $ git config --global user.email johndoe@example.com

!SLIDE smaller

# Getting Help #

    @@@ plain
    Man Pages
    
    $ git help <verb>
